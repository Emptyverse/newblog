2.匀速运动

2.1 js动画实现：position，marginLeft

position设置absolute，通过left来实现移动

2.2 简单动画存在的问题：

  • 边界处理：没有边界处理，动画会一直进行下去
  • 定时器的管理：多次触发动画会导致定时器累加，动画速度叠加

          padding: 0;

            height: 200px;
            background-color: pink;
            position: absolute;
            top: 50px;
        }
    </style>
</head>
<body>
    <!-- position marginLeft offsetLeft -->
    <button id="btn">动画启动</button>
    <div id="box"></div>
    <script>
        // 简单动画存在的问题： 1.边界处理，2.定时器管理
        // 1.获取事件源
        var btn = document.getElementById("btn");
        var box = document.getElementById("box");
        var timer = null;
        var num = 0;
        // 2.绑定按钮事件
        btn.onclick = function() {
            // 首先清除定时器 防止定时器累加，offsetLeft重叠累加
            setInterval(function() {
                num++;
                // box.style.marginLeft = num + 'px'
                box.style.left = num + 'px'
            }, 30);
            // timer = setInterval(function() {

            //     }

        }
    </script>
</body>

            function setAnimation(obj, end) {
                clearInterval(timer);
                speed = end > obj.offsetLeft ? 5 : -5;

                }, 5)
            }

4.缓动运动

速度由快到慢，变换透明度

<!DOCTYPE html>
<html lang="en">
<head>
   <meta charset="UTF-8">
   <meta http-equiv="X-UA-Compatible" content="IE=edge">
   <meta name="viewport" content="width=device-width, initial-scale=1.0">
   <title>Document</title>
   <style>
       #box {
           width: 200px;
           height: 200px;
           background-color: red;
           opacity: 0.5;
       }
   </style>
</head>
<body>
   <div id="box">

   </div>
   <script>
       window.onload = function() {
           var box = document.getElementById("box");
           var timer = null, alpha = 30, speed = 0;
           box.onmouseover = function() {
               opacityAnimation(this, 100);
           }
           function opacityAnimation(obj, end) {
               timer = setInterval(function() {
                   // 设置透明度变化速度
                   speed = end > alpha ? 1 : -1;
                   // 边界条件清除定时器
                   if(alpha === end) {
                       clearInterval(timer);
                       return;
                   }
                   // 修改当前alpha值
                   alpha += speed;
                   // 修改透明度
                   obj.style.opacity = alpha / 100; 
               }, 30)
           }
       }
   </script>
</body>
</html>
